export class BootScene extends Phaser.Scene {
  constructor() {
    super('BootScene');
  }

  preload() {}

  create() {
    const g = this.add.graphics();

    const makeTex = (key, drawFn, w=16, h=16) => {
      g.clear();
      drawFn(g);
      g.generateTexture(key, w, h);
    };

    // Tiles
    makeTex('tile', (gr) => {
      gr.fillStyle(0x2a2540, 1).fillRect(0,0,16,16);
      gr.fillStyle(0x201c33, 1).fillRect(0,0,16,4);
      gr.fillStyle(0x3a3358, 1).fillRect(0,4,16,12);
      gr.fillStyle(0x151124, 1).fillRect(2,6,3,3);
      gr.fillStyle(0x151124, 1).fillRect(11,10,3,3);
    });

    makeTex('brick', (gr) => {
      gr.fillStyle(0x3a2c23, 1).fillRect(0,0,16,16);
      gr.fillStyle(0x2a2019, 1).fillRect(0,0,16,3);
      gr.lineStyle(1, 0x1a1410, 1);
      for (let y=3; y<16; y+=5) {
        gr.beginPath(); gr.moveTo(0,y); gr.lineTo(16,y); gr.strokePath();
      }
      gr.beginPath(); gr.moveTo(8,3); gr.lineTo(8,8); gr.strokePath();
      gr.beginPath(); gr.moveTo(4,8); gr.lineTo(4,13); gr.strokePath();
      gr.beginPath(); gr.moveTo(12,8); gr.lineTo(12,13); gr.strokePath();
    });

    // Player
    makeTex('player', (gr) => {
      gr.fillStyle(0x0d0b14, 1).fillRect(3,2,10,13);
      gr.fillStyle(0x6bd6ff, 1).fillRect(5,4,6,3); // visor
      gr.fillStyle(0xffd36b, 1).fillRect(4,9,8,5); // body
      gr.fillStyle(0x0d0b14, 1).fillRect(4,14,3,2);
      gr.fillStyle(0x0d0b14, 1).fillRect(9,14,3,2);
    });

    // Gem
    makeTex('gem', (gr) => {
      gr.fillStyle(0x2d0b40, 1).fillRect(0,0,16,16);
      gr.fillStyle(0x000000, 0).fillRect(0,0,16,16);
      gr.fillStyle(0x9f4dff, 1).fillTriangle(8,1, 14,7, 8,15);
      gr.fillStyle(0x6a2bd8, 1).fillTriangle(8,1, 2,7, 8,15);
      gr.fillStyle(0xe7c7ff, 1).fillRect(9,4,2,3);
    });

    // Key
    makeTex('key', (gr) => {
      gr.fillStyle(0x000000, 0).fillRect(0,0,16,16);
      gr.fillStyle(0xf5d66a, 1).fillRect(3,7,8,3);
      gr.fillStyle(0xf5d66a, 1).fillRect(9,6,3,5);
      gr.fillStyle(0x9b7b23, 1).fillRect(4,8,6,1);
      gr.fillStyle(0xf5d66a, 1).fillCircle(4,8,3);
      gr.fillStyle(0x0b0b12, 1).fillCircle(4,8,1);
    });

    // Exit Door
    makeTex('exit', (gr) => {
      gr.fillStyle(0x1b152a, 1).fillRect(0,0,16,16);
      gr.fillStyle(0x4ad66b, 1).fillRect(3,2,10,12);
      gr.fillStyle(0x2a8f40, 1).fillRect(4,3,8,10);
      gr.fillStyle(0x0b0b12, 1).fillCircle(11,9,1);
    });

    // Spikes
    makeTex('spikes', (gr) => {
      gr.fillStyle(0x000000, 0).fillRect(0,0,16,16);
      gr.fillStyle(0xd65555, 1);
      for (let i=0;i<4;i++){
        const x=i*4;
        gr.fillTriangle(x,16, x+2,6, x+4,16);
      }
      gr.fillStyle(0x7d1f1f, 1).fillRect(0,14,16,2);
    });

    // Saw
    makeTex('saw', (gr) => {
      gr.fillStyle(0x000000, 0).fillRect(0,0,16,16);
      gr.fillStyle(0xcfd3de, 1).fillCircle(8,8,7);
      gr.fillStyle(0x7b8596, 1).fillCircle(8,8,5);
      gr.fillStyle(0x2b2f3a, 1).fillCircle(8,8,2);
      // teeth
      gr.fillStyle(0xe9edf6, 1);
      gr.fillRect(8,0,1,3);
      gr.fillRect(8,13,1,3);
      gr.fillRect(0,8,3,1);
      gr.fillRect(13,8,3,1);
    });

    // Enemy mummy
    makeTex('mummy', (gr) => {
      gr.fillStyle(0x000000, 0).fillRect(0,0,16,16);
      gr.fillStyle(0xc9c2a8, 1).fillRect(4,3,8,11);

      gr.fillStyle(0xc9c2a8, 1).fillRect(11,9,2,3);

    // Tiny particle
    makeTex('puff', (gr) => {
      gr.fillStyle(0x000000, 0).fillRect(0,0,8,8);
      gr.fillStyle(0xffffff, 1).fillCircle(4,4,3);
      gr.fillStyle(0xffffff, 0.4).fillCircle(3,3,3);
    }, 8, 8);

    this.scene.start('GameScene');
  }
}



const TILE = 16;

const LEVEL = [
  "############################################################",
  "#..........................................................#",
  "#.............g...............^.............g..............#",
  "#.....................#######...........#####..............#",
  "#..............####..................m.....................#",
  "#.....k............................#####.............e.....#",
  "#.............#######...................#######............#",
  "#..........................s...............................#",
  "#..........#####..................#####....................#",
  "#..........................................................#",
  "############################################################",
];

export class GameScene extends Phaser.Scene {
  constructor() {
    super('GameScene');
  }

  create() {
    this.cameras.main.setZoom(2);
    this.cameras.main.setBounds(0, 0, LEVEL[0].length * TILE, LEVEL.length * TILE);

    // World groups
    this.solids = this.physics.add.staticGroup();
    this.gems = this.physics.add.staticGroup();
    this.keys = this.physics.add.staticGroup();
    this.spikes = this.physics.add.staticGroup();
    this.exit = this.physics.add.staticGroup();
    this.enemies = this.physics.add.group({ allowGravity: true, immovable: false });
    this.hazards = this.physics.add.group({ allowGravity: false, immovable: true });

    // Build level
    this.spawn = { x: 3*TILE, y: 8*TILE };

    for (let y=0; y<LEVEL.length; y++) {
      for (let x=0; x<LEVEL[y].length; x++) {
        const ch = LEVEL[y][x];
        const wx = x*TILE + TILE/2;
        const wy = y*TILE + TILE/2;

        if (ch === '#') {
          this.solids.create(wx, wy, (y===0 || y===LEVEL.length-1) ? 'brick' : 'tile');
        } else if (ch === 'g') {
          this.gems.create(wx, wy, 'gem');
        } else if (ch === 'k') {
          this.keys.create(wx, wy, 'key');
        } else if (ch === '^') {
          this.spikes.create(wx, wy, 'spikes');
        } else if (ch === 'e') {
          this.exit.create(wx, wy, 'exit');
        } else if (ch === 'm') {
          const m = this.enemies.create(wx, wy, 'mummy');
          m.body.setSize(12, 14).setOffset(2, 2);
          m.setCollideWorldBounds(false);
          m.setBounce(0);
          m.setDragX(0);
          m.setMaxVelocity(80, 500);
          m.patrolDir = -1;
        } else if (ch === 's') {
          // moving saw hazard
          const s = this.hazards.create(wx, wy, 'saw');
          s.body.setSize(14,14).setOffset(1,1);
          s.originX = wx;
          s.t = 0;
        }
      }
    }

    // Player
    this.player = this.physics.add.sprite(this.spawn.x, this.spawn.y, 'player');
    this.player.body.setSize(10, 14).setOffset(3, 2);
    this.player.setMaxVelocity(220, 700);
    this.player.setDragX(1200);

    // Colliders
    this.physics.add.collider(this.player, this.solids);
    this.physics.add.collider(this.enemies, this.solids);

    // Overlaps
    this.physics.add.overlap(this.player, this.gems, this.onGem, null, this);
    this.physics.add.overlap(this.player, this.keys, this.onKey, null, this);
    this.physics.add.overlap(this.player, this.spikes, () => this.killPlayer('spikes'), null, this);
    this.physics.add.overlap(this.player, this.hazards, () => this.killPlayer('saw'), null, this);
    this.physics.add.overlap(this.player, this.enemies, () => this.killPlayer('mummy'), null, this);
    this.physics.add.overlap(this.player, this.exit, this.onExit, null, this);

    // Input
    this.cursors = this.input.keyboard.createCursorKeys();
    this.keyR = this.input.keyboard.addKey(Phaser.Input.Keyboard.KeyCodes.R);

    // Jump tech
    this.coyoteTimeMs = 90;
    this.jumpBufferMs = 110;
    this.lastOnGroundAt = 0;
    this.jumpPressedAt = -9999;

    // State
    this.score = 0;
    this.hasKey = false;
    this.dead = false;

    // Particles
    this.particles = this.add.particles(0, 0, 'puff', {
      speed: { min: 20, max: 90 },
      lifespan: { min: 180, max: 320 },
      scale: { start: 0.7, end: 0 },
      quantity: 6,
      emitting: false
    });

    // UI
    this.ui = this.add.text(6, 6, '', {
      fontFamily: 'monospace',
      fontSize: '14px',
      color: '#cfe3ff'
    }).setScrollFactor(0).setDepth(999);

    this.hint = this.add.text(6, 26, 'Arrows: move/jump • R: restart', {
      fontFamily: 'monospace',
      fontSize: '12px',
      color: '#8aa0c7'
    }).setScrollFactor(0).setDepth(999);

    // Camera follow
    this.cameras.main.startFollow(this.player, true, 0.12, 0.12);

    // A little “Flash pop” at start
    this.cameras.main.flash(200, 20, 20, 35);

    this.updateUI();
  }

  update(time, dt) {
    if (Phaser.Input.Keyboard.JustDown(this.keyR)) {
      this.scene.restart();
      return;
    }
    if (this.dead) return;

    const body = this.player.body;

    // Track ground contact for coyote time
    if (body.blocked.down) this.lastOnGroundAt = time;

    // Capture jump press for buffer
    if (Phaser.Input.Keyboard.JustDown(this.cursors.up) || Phaser.Input.Keyboard.JustDown(this.cursors.space)) {
      this.jumpPressedAt = time;
    }

    // Horizontal movement
    const left = this.cursors.left.isDown;
    const right = this.cursors.right.isDown;

    const accel = 1200;
    if (left) {
      body.setAccelerationX(-accel);
      this.player.setFlipX(true);
    } else if (right) {
      body.setAccelerationX(accel);
      this.player.setFlipX(false);
    } else {
      body.setAccelerationX(0);
    }

    // Jump evaluation using coyote time + buffer
    const canCoyote = (time - this.lastOnGroundAt) <= this.coyoteTimeMs;
    const buffered = (time - this.jumpPressedAt) <= this.jumpBufferMs;

    if (buffered && canCoyote) {
      body.setVelocityY(-520);
      this.jumpPressedAt = -9999;
      this.puff(this.player.x, this.player.y + 8);
      this.cameras.main.shake(60, 0.002);
    }

    // Variable jump height (cut jump if released early)
    const jumpHeld = this.cursors.up.isDown || this.cursors.space.isDown;
    if (!jumpHeld && body.velocity.y < -120) {
      body.setVelocityY(body.velocity.y * 0.88);
    }

    // “Flashy” quick fall
    if (body.velocity.y > 0) {
      body.setGravityY(2200);
    } else {
      body.setGravityY(1600);
    }

    // Enemy patrol
    this.enemies.children.iterate((e) => {
      if (!e?.body) return;
      // Simple edge flip via wall detection
      if (e.body.blocked.left) e.patrolDir = 1;
      if (e.body.blocked.right) e.patrolDir = -1;
      e.setVelocityX(55 * e.patrolDir);
    });

    // Hazards (saw moves side to side)
    this.hazards.children.iterate((h) => {
      if (!h?.body) return;
      h.t += dt * 0.003;
      h.x = h.originX + Math.sin(h.t) * 48;
      h.rotation += dt * 0.01;
      h.body.reset(h.x, h.y);
    });

    this.updateUI();
  }

  puff(x, y) {
    this.particles.setPosition(x, y);
    this.particles.explode(Phaser.Math.Between(5, 8));
  }

  updateUI() {
    this.ui.setText(
      `Score: ${this.score}   Key: ${this.hasKey ? 'YES' : 'NO'}   Gems left: ${this.gems.countActive(true)}`
    );
  }

  onGem(player, gem) {
    gem.destroy();
    this.score += 10;
    this.puff(gem.x, gem.y);
    this.cameras.main.shake(40, 0.002);
  }

  onKey(player, key) {
    key.destroy();
    this.hasKey = true;
    this.score += 50;
    this.puff(key.x, key.y);
    this.cameras.main.shake(70, 0.003);
  }

  onExit(player, exit) {
    if (!this.hasKey) {
      // little nudge feedback
      this.cameras.main.shake(60, 0.002);
      return;
    }
    // Win
    this.dead = true;
    this.puff(exit.x, exit.y);
    this.cameras.main.flash(250, 60, 180, 80);
    const msg = this.add.text(this.cameras.main.scrollX + 40, this.cameras.main.scrollY + 60,
      "ESCAPED!\nPress R to restart",
      { fontFamily:'monospace', fontSize:'18px', color:'#e7c7ff' }
    ).setDepth(1000);

    msg.setShadow(2,2,'#000',2);
  }

  killPlayer(why) {
    if (this.dead) return;
    this.dead = true;

    this.puff(this.player.x, this.player.y);
    this.player.setTint(0xff6b6b);
    this.player.setVelocity(Phaser.Math.Between(-120, 120), -260);

    this.cameras.main.shake(140, 0.006);
    this.cameras.main.flash(140, 90, 20, 20);

    const msg = this.add.text(this.cameras.main.scrollX + 40, this.cameras.main.scrollY + 60,
      `YOU DIED (${why.toUpperCase()})\nPress R to restart`,
      { fontFamily:'monospace', fontSize:'16px', color:'#ffd6d6' }
    ).setDepth(1000);

    msg.setShadow(2,2,'#000',2);
  }
}
